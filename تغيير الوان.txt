-- Dropdown لاختيار اللون
local ColorDropdown = SettingsTab:AddDropdown({
    Name = "اختر اللون",
    Default = "وردي فاتح",
    Options = (function()
        local t = {}
        for _,c in ipairs(Colors) do
            table.insert(t, c.Name)
        end
        return t
    end)(),
    Callback = function(Value)
        for _,c in ipairs(Colors) do
            if c.Name == Value then
                SelectedColor = c.Color
            end
        end
    end
})

-- زر لتفعيل اللون المختار
SettingsTab:AddButton({
    Name = "تفعيل اللون",
    Callback = function()
        -- تحديث الثيم ديناميكياً
        local function UpdateThemeColor(color)
            redzlib.Themes.Darker["Color Hub 1"] = ColorSequence.new({
                ColorSequenceKeypoint.new(0.00, color),
                ColorSequenceKeypoint.new(0.50, color),
                ColorSequenceKeypoint.new(1.00, color)
            })
            redzlib.Themes.Darker["Color Hub 2"] = color
            redzlib.Themes.Darker["Color Stroke"] = color
            redzlib.Themes.Darker["Color Theme"] = color
            redzlib.Themes.Darker["Color Text"] = color
            redzlib.Themes.Darker["Color Dark Text"] = color

            -- تحديث فورياً
            pcall(function()
                if Library and Library.SetTheme then
                    Library:SetTheme("Darker")
                end
                if Library and Library.Update then
                    Library:Update()
                end
                if Window and Window.Refresh then
                    Window:Refresh()
                end
            end)
        end

        UpdateThemeColor(SelectedColor)
    end
})

local RainbowEnabled = false

    Callback = function(Value)
        RainbowEnabled = Value
        spawn(function()
            while RainbowEnabled do
                for i = 0, 1, 0.01 do

                    pcall(function()
                        redzlib.Themes.Darker["Color Hub 1"] = ColorSequence.new({

                        redzlib.Themes.Darker["Color Theme"] = color

                        if Library and Library.Update then Library:Update() end
                        if Window and Window.Refresh then Window:Refresh() end
                    end)
                    task.wait(0.05)
                    if not RainbowEnabled then break end
                end
            end
        end)
    end
})
